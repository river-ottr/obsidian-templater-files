<%* // Generated by daily note templates.
const daily_format = "MM-DD-YYYY";
const weekly_format = "w YYYY";
let weekly_title = ""
if (/Week (\d|\d{2}) \d{4}/.test(tp.file.title)){ // listen event on file creation. regex is our protection against false flags. we'll assume the date from the "Week" WW YYYY format.
  weekly_title = tp.file.title;
}
else{ // not correctly named, we'll rename it based on today's date. this will fire when used as a template directly.
  weekly_title = "Week " + moment.format(weekly_format);
  await tp.file.rename(weekly_title);
} 
// create buttons to scroll weeks, considering the date from the note's title. 
const last_week_title = "Week " + moment(tp.file.title, weekly_format).subtract(1, "w").format(weekly_format);
const next_week_title = "Week " + moment(tp.file.title, weekly_format).add(1, "w").format(weekly_format); 
// days of the week as smart links, based on the note's title
const weekdaysArr = []
for(let i = 0; i <=6; i++){
  weekdaysArr[i] = moment(tp.file.title, weekly_format).week(tp.file.title.split(" ")[1]).day(i).format(daily_format);
}
